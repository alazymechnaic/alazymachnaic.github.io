# 无源码测试基于qemu实现，首先配置qemu
$ cd qemu_mode
$ ./build_qemu_support.sh
# 设置原始输入和输出
$ mkdir fuzz_in
$ echo "hello" > fuzz_in/testcase
$ gcc test.c -o test
$ afl-fuzz -i fuzz_in -o fuzz_out -Q ./test
