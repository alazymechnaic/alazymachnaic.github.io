表格

插入表格并且让其占据两栏

\usepackage{stfloats}
\begin{table*}[ht]
    % 插入长度为5pt的垂直空间（也可以是负数，缩进）
    \vspace{5pt}
    \centering
    % 表名 前面为中文名/后面为英文名
    \caption{Total Number Display Table}
    % label标签，用以引用本表时。例：autoref{num}
    \label{num}
    % 设置表格单元格的列宽
    \setlength{\tabcolsep}{6mm}{
    % 表示 三线表 有4列
    \begin{tabular}{cccc}
    % toprule表示三线表的顶部线
        \toprule
        Program & CVE & Primitive Location  & Number of Tainted instructions \\
        % midrule 表示 三线表的 中部线
        \midrule
        % 合并三行1列，用空格代替，也可以用\multirow{}[]{}{}来表示
        \textbf{Gstreamer}  & b  & 10   & b \\
        \textbf{ }    & 1   & 1   & 1   \\
        \textbf{ }    & 1    & 1   & 1   \\
   
        \textbf{latex2rtf}  & 1     & 2  & 1  \\
        \textbf{ }    & 1       & 3  & 1  \\
        \textbf{ }    & 1       & 1  & 1   \\  
        \textbf{ }    & 1       & 1  & 1   \\ 
    
        \textbf{nginx}  & 1     & 30   & 1   \\  
        \textbf{ }    & 1   & 50   & 1  \\
        \textbf{ }    & 1     & 100  & 1 \\
        \textbf{ }    & 1     & 80   & 1  \\
        \textbf{ }    & 1     & 40   & 1  \\
        
        \textbf{ALL}       & dd      & dd     & ddd \\
        % bottomrule表示 三线表 的底部线
        \bottomrule 
    \end{tabular}}
\end{table*}
